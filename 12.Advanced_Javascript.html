//Create two classes: an Animal class and a Mamal class.
//Create a cow that accepts a name, type and color and has a sound method that moo's her name, type and color.
class Animal {
constructor(name, type, color) {
this.name = name;
this.color = color;
this.type = type;
}
introduce(){
console.log(`Hello, My name is ${this.name}, I am a ${this.color} ${this.type}`)
}}
class Mamal extends Animal {
constructor(name, type, color) {
super(name, type, color)
}
sound() {
console.log(`Mooooo, My name is ${this.name}, I am a ${this.color} ${this.type}`);
}
}
const cow = new Mamal('Shelly', 'Milk Cow', 'White');
cow.introduce();
cow.sound();
